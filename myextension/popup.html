document.addEventListener("DOMContentLoaded", () => { const container = document.getElementById("imageContainer"); chrome.storage.local.get(null, data => { const keys = Object.keys(data).filter(key => key.startsWith("instagram_img_")); if (keys.length === 0) { container.innerText = "No images stored yet."; return; } container.innerHTML = ""; // Clear loading text keys.forEach(key => { const base64 = data[key]; const img = document.createElement("img"); img.src = base64; container.appendChild(img); }); }); });
